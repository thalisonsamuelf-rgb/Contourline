  # FRAMEWORK 3: 7 SWEEPS OF COPY EDITING
  # ─────────────────────────────────────────────────────────────────────────────
  seven_sweeps:
    name: "7 Sweeps of Copy Editing"
    category: "editing"
    origin: "Copyhackers (Joanna Wiebe) - Usado por Ry Schwartz"
    command: "*7-sweeps"

    philosophy: |
      A mágica do copy não acontece na escrita - acontece na edição.
      Cada sweep tem um foco específico. Não tente fazer tudo de uma vez.
      Passe pelo copy 7 vezes, cada vez com um único objetivo.

    the_sweeps:
      sweep_1:
        name: "Clarity Sweep"
        focus: "O copy é claro e fácil de entender?"
        questions:
          - "Cada frase comunica UMA ideia claramente?"
          - "Um leitor de 8ª série entenderia?"
          - "Há jargão desnecessário?"
        action: "Simplifique frases confusas, elimine jargão"

      sweep_2:
        name: "Voice & Tone Sweep"
        focus: "O copy soa como a marca/pessoa?"
        questions:
          - "Soa consistente do início ao fim?"
          - "Soa como a pessoa realmente fala?"
          - "O tom é apropriado para o momento?"
        action: "Ajuste voz para consistência e autenticidade"

      sweep_3:
        name: "So What Sweep"
        focus: "O leitor se importa com cada frase?"
        questions:
          - "Por que o leitor deveria se importar?"
          - "Cada claim está conectada a um benefício?"
          - "Há frases que não adicionam valor?"
        action: "Conecte cada afirmação ao que importa para o leitor"

      sweep_4:
        name: "Prove It Sweep"
        focus: "Cada claim tem suporte/prova?"
        questions:
          - "Há evidência para cada promessa?"
          - "Os testimonials são específicos?"
          - "Os números são verificáveis?"
        action: "Adicione prova específica para cada claim importante"

      sweep_5:
        name: "Specificity Sweep"
        focus: "O copy é específico o suficiente?"
        questions:
          - "Há números específicos ao invés de vagos?"
          - "Os exemplos são concretos?"
          - "As promessas são precisas?"
        action: "Substitua genérico por específico sempre que possível"

      sweep_6:
        name: "Heightened Emotion Sweep"
        focus: "O copy conecta emocionalmente?"
        questions:
          - "O leitor SENTE algo ao ler?"
          - "As emoções certas estão sendo evocadas?"
          - "Há momentos de conexão emocional?"
        action: "Adicione camadas emocionais onde apropriado"

      sweep_7:
        name: "Zero Risk Sweep"
        focus: "Todas as objeções e riscos foram endereçados?"
        questions:
          - "O leitor sabe exatamente o que esperar?"
          - "Os medos foram reconhecidos e resolvidos?"
          - "A garantia é clara e confiante?"
        action: "Remova todo risco percebido da decisão"

    application: |
      Use DEPOIS de escrever o primeiro draft.
      Faça um sweep completo por vez.
      Não pule sweeps - cada um pega algo diferente.

  # FRAMEWORK 5: CART CLOSE WITHOUT PRESSURE
  # ─────────────────────────────────────────────────────────────────────────────
  cart_close_framework:
    name: "Cart Close Without Pressure"
    category: "conversion"
    origin: "Ry Schwartz - Email Swipes Podcast"
    command: "*cart-close"

    philosophy: |
      "There is kind of a pressure of will I or will I not do it? That's already
      internally generated. Like in a prospect who is genuinely interested in it.
      They're already feeling some degree of conflict and pressure of wanting it,
      but not necessarily wanting to make an investment. Why would I add to the
      pressure? Pressure plus pressure equals overwhelm equals shutdown."

    the_insight: |
      O prospect interessado JÁ está sentindo pressão interna.
      Adicionar pressão externa causa shutdown, não conversão.
      Seu trabalho é ALIVIAR a pressão enquanto encoraja decisão.

    techniques:
      lighten_the_mood:
        description: "Use humor ou leveza apropriada"
        example: "I know your inbox is probably screaming at you right now..."
        why_it_works: "Reconhece a realidade e humaniza a comunicação"

      remove_decision_weight:
        description: "Faça a decisão parecer reversível"
        example: "Try it for 30 days. If it's not for you, no hard feelings."
        why_it_works: "Reduz o peso percebido da decisão"

      validate_their_hesitation:
        description: "Diga que é OK se eles não comprarem"
        example: "If the timing isn't right, I genuinely want you to wait."
        why_it_works: "Remove a sensação de ser 'vendido'"

      focus_on_their_best_interest:
        description: "Faça ser sobre ELES, não sobre você"
        example: "Whatever you decide, I want it to be the right decision for YOU."
        why_it_works: "Constrói confiança e remove defensividade"

      recap_without_hype:
        description: "Relembre o valor de forma factual"
        example: "Here's what you get: [list]. The cohort starts [date]."
        why_it_works: "Informa sem manipular"

    anti_patterns:
      never_do:
        - "LAST CHANCE!!!" (all caps urgency)
        - "You'll regret missing this" (guilt trip)
        - "Everyone else is joining" (FOMO manipulation)
        - "I can't believe you haven't signed up yet" (shame)
        - Countdown timers every email (pressure overload)

    application: |
      Use nos últimos 24-48 horas de um lançamento.
      O objetivo é facilitar a decisão, não forçá-la.
      Se o prospect não está pronto, deixe ir com graça.
